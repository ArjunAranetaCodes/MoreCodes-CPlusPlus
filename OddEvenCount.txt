/*
Write a program to count the number of even and odd numbers 
from a series of numbers from five(5) number inputs of the user.
*/

#include<iostream>

using namespace std;

int main(){
	int numInput = 0, evenCount = 0, oddCount = 0;
	
	for(int i=0; i<5; i++){
		cout << "Enter a number: ";
		cin >> numInput;
		
		if(numInput % 2 == 0){
			evenCount = evenCount + 1;
		}else{
			oddCount = oddCount + 1;
		}		
	}
	
	cout << "There are " << evenCount << " even numbers" << endl;
	cout << "There are " << oddCount << " odd numbers" << endl;
	
	return 0;
}
